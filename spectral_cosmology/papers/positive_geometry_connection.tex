\subsection{The Self-Reference Polytope}

\begin{definition}[Self-Reference Polytope] \label{def:psr}
\textsc{[Conjectured]}

Let $\{\lambda_k\}_{k=0}^{\infty}$ be the spectrum of the spatial Laplacian. The \emph{self-reference polytope} $P_{SR}$ is defined by the constraints:
\begin{enumerate}
    \item \textbf{Connectivity}: $\lambda_1 > 0$ (connected manifold)
    \item \textbf{Integration bound}: $\lambda_1 \leq c/L^2$ (finite-time integration)
    \item \textbf{Error tolerance}: $\lambda_1 > \Gamma = \epsilon_0/\tau$ (stable self-modeling)
    \item \textbf{Weyl asymptotics}: $\lambda_k \sim k^{2/n}$ for large $k$
    \item \textbf{Spectral gap}: $\delta = \lambda_2 - \lambda_1 > \delta_{min}$ (perturbation stability)
\end{enumerate}
\end{definition}

\begin{theorem}[Non-Empty $P_{SR}$ Implies Cosmological Structure]
\textsc{[Derived]}

If the self-reference polytope $P_{SR}$ is non-empty, then:
\begin{enumerate}
    \item The spatial manifold must be compact with positive curvature
    \item The first eigenvalue satisfies $\Gamma < \lambda_1 \lesssim 1/L^2$
    \item For de Sitter space with $\lambda_1 = \Lambda$: we obtain $\Lambda \sim H^2$
\end{enumerate}
\end{theorem}

\begin{proof}
Constraints (1)-(3) in Definition \ref{def:psr} directly imply:
\[
0 < \Gamma < \lambda_1 \leq \frac{c}{L^2}
\]

For this range to be non-empty, we require $\Gamma < c/L^2$, i.e., $\epsilon_0/\tau < c/L^2$.

For de Sitter space, $\lambda_1 = \Lambda$ and $L \sim 1/H$ (Hubble radius). Thus:
\[
\Gamma < \Lambda \lesssim c \cdot H^2
\]

The upper bound gives $\Lambda \sim H^2$. The lower bound $\Lambda > \Gamma$ ensures self-modeling remains stable.
\end{proof}

\subsection{Connection to Dimensionality}

The proof that $n = 3$ spatial dimensions from self-reference (Theorem \ref{thm:dimension} in dimensionality section) acquires new meaning:

\begin{proposition}[Dimension as Polytope Constraint]
\textsc{[Derived]}

The dimensionality constraint $n \leq 3$ is equivalent to requiring $P_{SR}$ to be non-empty at all scales $L$.
\end{proposition}

\begin{proof}
From the dimensionality theorem, self-reference requires $L^2 \geq c \cdot L^{n-1}$ at all scales. This is the constraint:
\[
L^{3-n} \geq c
\]

For $n > 3$: As $L \to \infty$, $L^{3-n} \to 0$, violating the bound. The polytope becomes empty at large scales.

For $n \leq 3$: The constraint is satisfied for all $L \geq c^{1/(3-n)}$ (or all $L$ if $n = 3$).

Thus $n = 3$ is the unique dimension where $P_{SR}$ is non-empty at all scales.
\end{proof}
